\subsection[static]{Static members}

\begin{frame}[fragile]
  \frametitlecpp[98]{Static members}
  \begin{block}{Concept}
    \begin{itemize}
    \item members attached to a class rather than to an object
    \item usable with or without an instance of the class
    \item identified by the \cppinline{static} keyword
    \end{itemize}
  \end{block}

  \vspace{-1\baselineskip}
  \begin{overprint}
  \onslide<1>
  \begin{exampleblock}{Static.hpp}
  \begin{cppcode}
    struct AddNumbers {
        int localI;
        inline static int staticI = 0; // inline is c++17

        int Get() const { return staticI + localI; }
        static void SetStatic(int i) { staticI = i; }
    };
  \end{cppcode}
  \end{exampleblock}

  \onslide<2>
  \begin{exampleblockGB}{Static.cpp}{https://godbolt.org/z/KjvzrTnP1}{Static members}
  \begin{cppcode}
    AddNumbers::SetStatic(1);

    AddNumbers a{1}; AddNumbers b{5};
    a.Get(); // 2
    b.Get(); // 6

    a.SetStatic(100);
    a.Get(); // 101
    b.Get(); // 105
  \end{cppcode}
  \end{exampleblockGB}
  \end{overprint}
\end{frame}
